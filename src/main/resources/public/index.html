/users

/users/login (GET) - вход в аккаунт
login: String - логин
password: String - пароль

/users/get (GET) - пользователь по логину
login: String - логин

/users/register (POST) - регистрация пользователя
fName: String - имя
lName: String - фамилия
login: String - логин
password: String - пароль
photo: String - аватар


/news

/news (GET) - список всех новостей

/news/post (POST) - добавить новость
title: String - название новости
content: String - текст новости
image: String - картинка к новости
userLogin: String - логин пользователя, который добавил новость

/news/comments (GET) - комментарии к новости
newId: Int - ID новости

/news/comments/post (POST) - обновление геопозиции
newId: Int - ID новости
userLogin: String - логин автора комментария
content: String - текст комментария

code: -1 http code: 500 undefined error
code: 1 http code: 500 query param $paramName undefined
code: 2 http code: 500 post param $paramName undefined
code: 3 http code: 400 query param $paramName in not $type
code: 4 http code: 400 query param $paramName incorrect
code: 5 http code: 404 $what not found
